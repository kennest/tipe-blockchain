\begin{frame} \frametitle{Proof of work}
\begin{itemize}
\item \textit{Hash} du bloc dépend du contenu et du \textit{nonce} ; doit être inférieur à une difficulté
\item Donc nécessité de tester tous les nonces un à un 
\item Exemples : 
	\begin{itemize}
	%\item \textcolor{pale}{Alice envoie 10 BTC à un marchand, retarde l'heure de son ordinateur, puis renvoie ces 10 BTC à elle-même}
	\item  Alice envoie 10 BTC à un marchand, retarde l'heure de son ordinateur, puis renvoie ces 10 BTC à elle-même
	\item Alice doit alors lutter seule contre la puissance générale du réseau lors de la \textit{proof of work}
	\end{itemize}
\item  Incitation en récompensant les mineurs par 25 BTC
\end{itemize}
\end{frame}

\begin{frame} \frametitle{Questionnements}
\begin{itemize}
\item \textit{Proof of Work} très gourmande en énergie
\item \textit{Blockchain} lourde (71 Go en juin 2016)
\item Croissance exponentielle de la \textit{blockchain} et du nombre de calculs à réaliser lors de la PoW
\item[]
\item \textit{Proof of Stake} : fonction de vérification qui nécessite moins de calculs car se base sur des critères comme le solde de l'utilisateur ou son ancienneté selon la monnaie
\end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame} \frametitle{Conclusion}
\begin{itemize}
\item Premiers résultats
	\begin{itemize}
		\item Diffusion plus importante si la difficulté est plus élevée
	\item Dans notre modèle, si la puissance d'un seul utilisateur est plus élevée que celle des autres, sa \textit{blockchain} a tendance à prédominer
		\end{itemize}
	\item Suites du travail 
		\begin{itemize}
		\item Comparer la \textit{Proof of Work} et la \textit{Proof of Stake}
		\item Quels sont les paramètres de PoS les plus économes tout en conservant la sécurité de la PoW
		\end{itemize}
\end{itemize} 
\end{frame}
