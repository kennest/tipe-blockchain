\begin{frame} \frametitle{Principe de diffusion de l'information}
  \begin{itemize}
  \only<1>{\item Au départ, un nœud a une information}
  \only<2->{\item Itérations %jusqu'à ce que tous les nœuds aient une information
    \begin{itemize}
    \item<2-> Parcours des nœuds un à un, chacun passe son information à ses voisins qui ne la possèdent pas
    \only<2>{\item Si le voisin est \emph{normal}, l'information passe inchangée}
    \only<3>{\item Si il est \emph{attaquant}, l'information passe à Faux}
    \only<4>{\item Fin lorsque tous les nœuds ont reçu une information}
    \end{itemize}}
  \end{itemize}

\begin{frame} \frametitle{Les types de réseaux - Réseau invariant d'échelle}
% Images !
  \begin{itemize}
  \item \emph{Scale-free}, ou invariant d'échelle
    \begin{itemize}
      \item La probabilité qu'un noeud ait $k$ voisins est $\alpha k^{-\gamma}$
      \item Le réseau Internet est de ce type.
    \end{itemize}
  \item Nous utiliserons l'algorithme de Barabási-Albert, pour lequel $\gamma = 3$ et $\alpha \approx 0.83$.
  \end{itemize}

\begin{frame} \frametitle{Plan d'expérience}
  \begin{itemize}
  \item Pour chaque jeu de paramètres, on effectue $k$ simulations
  \item Une simulation consiste en
    \begin{itemize}
    \item Génération d'un réseau
          \begin{itemize}
          \item Variation du nombre d'attaquants $n$
          \item Diffusion de l'information pour chaque valeur de $n$
          \end{itemize}
    \item Résultat de la simulation : proportion d'informations fausses en fonction du nombre d'attaquants
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Sur un réseau homogène}

\begin{frame} \frametitle{Variation du nombre de nœuds :  50 et 100 noeuds}

  %\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-10-3-5-i100.png}
  \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-2-i100.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-100-3-1-i100.png} 

  \begin{itemize}
  \item Part d'informations fausses selon le nombre d'attaquants
  \item Croissant, mais la courbe est concave : plus il y a d'attaquants, et moins l'action d'en ajouter un nouveau a un effet important
  \item Plus il y a de nœuds et plus le réseau est vulnérable
\end{itemize}
\end{frame}



\begin{frame} \frametitle{Variation du nombre d'arêtes par nœud}
\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-2-i100.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-20-2-i100.png} 

\begin{itemize}
	\item Courbe se divise en deux parties, la seconde affine
	\item A l'arrivée sur la partie affine, les seuls nœuds ayant des informations vraies sont voisins de l'émetteur
\end{itemize}
\end{frame}

\subsection{Sur un réseau invariant d'échelle}

\begin{frame}[fragile] \frametitle{Sur un réseau invariant d'échelle}

\end{itemize}

\section{Conclusion}

\begin{frame} \frametitle{Conclusion}
  \begin{itemize}
  \item Résultats
    \begin{itemize}
    \item Courbes non linéaires, croissantes
    \item Pour les graphes homogènes, division en deux parties, l'une affine
    \item Réseaux invariants d'échelle : présence de paliers
    \end{itemize}
  \item En pratique
    \begin{itemize}
    \item Réseau homogène difficile à mettre en place
    \item Dans le réseau scale-free, on a supposé que les noeuds ayant une forte connectivité ont la même probabilité d'être attaquants que les autres. Or en pratique, ce sont souvent des noeuds plus sécurisés.
    \end{itemize}
  \end{itemize} 
\end{frame}
