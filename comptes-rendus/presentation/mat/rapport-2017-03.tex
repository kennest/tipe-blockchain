\begin{frame} \frametitle{Principe de diffusion de l'information}
  \begin{itemize}
  \item Au départ, un agent a une information
  \item Itérations jusqu'à ce que tous les nœuds aient une information
    \begin{itemize}
    \item Parcours des agents un à un
    \item Si \emph{normal} : passe son information à tous les voisins qui ne la possèdent pas
    \item Si \emph{attaquant} : envoie une information fausse à tous ses voisins qui ne la possèdent pas encore
    \end{itemize}
  \end{itemize}

\begin{itemize}
\item Initialement, A possède une information vraie
\item Itération 1 : B et C reçoivent l'information vraie (B la modifie en faux)
\item Itération 2 : D reçoit l'information, fausse, en provenance de B
\end{itemize}
\end{frame}

\begin{frame} \frametitle{Les types de réseaux - Réseau invariant d'échelle}
% Images !
\begin{itemize}
\item \emph{Scale-free}, ou invariant d'échelle : Le nombre d'arêtes par noeud suit une loi de puissance :la probabilité qu'un noeud ait $k$ voisins est $k^{-\gamma}$ Le réseau Internet est de ce type.
\item Nous utiliserons l'algorithme de Barabási-Albert, pour lequel $\gamma = 3$.
\end{itemize}

\begin{frame} \frametitle{Plan d'expérience}
  \begin{itemize}
  \item Pour chaque jeu de paramètres, on effectue $k$ simulations
  \item Une simulation consiste en
    \begin{itemize}
    \item Génération d'un réseau, puis variation du nombre d'attaquants
    \item Résultat de la simulation : \% d'informations fausses en fonction du nombre d'attaquants
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Sur un réseau homogène}

\begin{frame} \frametitle{Variation du nombre de noeuds : 10 et 50 noeuds}

\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-10-3-4.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-1.png} 

\begin{itemize}
\item Nombre de réponses fausses selon le nombre d'attaquants
\item Croissant, mais la courbe est concave : plus il y a d'attaquants, et moins l'action d'en ajouter un nouveau est significative
\end{itemize}
\end{frame}



\begin{frame} \frametitle{Variation du nombre d'arêtes par agent}
\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-1.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-20-1.png} 

\begin{itemize}
	\item Variation du nombre d'arêtes par agent
	\item Courbe se divise en deux parties, la seconde affine
	\item A l'arrivée sur la dernière partie, les seuls agents ayant des informations vraies sont voisins de l'émetteur
\end{itemize}
\end{frame}

\subsection{Sur un réseau invariant d'échelle}

\begin{frame}[fragile] \frametitle{Sur un scale-free network}

\section{Conclusion}

\begin{frame} \frametitle{Conclusion}
\begin{itemize}
	\item Résultats
	\begin{itemize}
		\item Courbes non linéaires, croissantes
		\item Pour les graphes homogènes, division en deux parties affines
		\item Réseaux invariants d'échelle : présence de paliers
	\end{itemize}
	\item En pratique
	\begin{itemize}
		\item Le réseau en étoile est peu vulnérable, si le noeud central est non-attaquant
		\item Réseau homogène difficile à mettre en place
		\item Dans le réseau scale-free, on a supposé que les noeuds ayant une forte connectivité ont la même probabilité d'être attaquants que les autres. Or en pratique, ce sont souvent des noeuds plus sécurisés.
	\end{itemize}
\end{itemize} 
\end{frame}
