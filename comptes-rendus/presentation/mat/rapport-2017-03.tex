\section{Modélisation}

\begin{frame} \frametitle{Les objets du réseau}
\begin{itemize}
\item Agent : \texttt{(id, strategie, informations)}
\item Tunnel : \texttt{(emetteur récepteur)}
\item Information : \texttt{(id, destinataire, texte, passeurs)}
\item Réseau : \texttt{(liste d'agents, liste de tunnels)}
\vspace*{-0.4cm} 
\end{itemize}
\end{frame}


\begin{frame} \frametitle{Fonctionnement général}
\begin{itemize}
\item Fonctionnement multi-agents, en effectuant de multiples \emph{itérations} sur le réseau
\item Itération :
	\begin{itemize}
	\item Parcours des agents un à un
	\item Si \emph{normal} : passe son information à tous les voisins qui ne la possèdent pas
	\item Si \emph{attaquant} : envoie à tous ses voisins qui ne possèdent pas encore l'information une information de même \texttt{id}, mais fausse
	\end{itemize}
\end{itemize}
\end{frame}


\begin{frame} \frametitle{Les types de réseaux - En étoile}
% Images !
\begin{itemize}
\item \emph{En étoile} : Un noeud est connecté avec tous les autres. C'est l'architecture d'un système client-serveur.
\end{itemize}
\begin{center}
\includegraphics[scale=0.2]{star-network.png}
\end{center}
\end{frame}


\begin{frame} \frametitle{Les types de réseaux - Homogène}
% Images !
\begin{itemize}
\item \emph{Homogène} : Chaque noeud possède un même nombre \texttt{nb\_tun} d'arètes pointant vers un autre noeud.
\end{itemize}

\begin{center}
\includegraphics[scale=0.4]{homogene.png}
\end{center}
\end{frame}


\begin{frame} \frametitle{Les types de réseaux - Reseau invariant d'échelle}
% Images !
\begin{itemize}
\item \emph{Scale-free, ou invariant d'échelle} : Le nombre d'arètes par noeud suit une loi de puissance : $ P(k) \sim k^{-\gamma}$. Le réseau Internet est de ce type.
\end{itemize}
\begin{center}
\includegraphics[scale=0.25]{scale-free.png}
\includegraphics[scale=0.15]{power-law.png} 
\end{center}
\end{frame}

\end{itemize}
