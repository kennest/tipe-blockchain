\section{Modélisation}

\begin{frame} \frametitle{Les objets du réseau}
\begin{itemize}
\item Agent : \texttt{(id, strategie, informations)}
\item Tunnel : \texttt{(emetteur récepteur)}
\item Information : \texttt{(id, destinataire, texte, passeurs)}
\item Réseau : \texttt{(liste d'agents, liste de tunnels)}
\vspace*{-0.4cm} 
\end{itemize}
\end{frame}


\begin{frame} \frametitle{Fonctionnement général}
\begin{itemize}
\item Fonctionnement multi-agents, en effectuant de multiples \emph{itérations} sur le réseau
\item Itération :
	\begin{itemize}
	\item Parcours des agents un à un
	\item Si \emph{normal} : passe son information à tous les voisins qui ne la possèdent pas
	\item Si \emph{attaquant} : envoie à tous ses voisins qui ne possèdent pas encore l'information une information de même \texttt{id}, mais fausse
	\end{itemize}
\end{itemize}
\end{frame}


\begin{frame} \frametitle{Les types de réseaux}
\begin{itemize}
\item \emph{En étoile} : Un noeud est connecté avec tous les autres
\item \textit{Blockchain} lourde (71 Go en juin 2016)
\item Croissance exponentielle de la \textit{blockchain} et du nombre de calculs à réaliser lors de la PoW
\item[]
\item \textit{Proof of Stake} : fonction de vérification qui nécessite moins de calculs car se base sur des critères comme le solde de l'utilisateur ou son ancienneté selon la monnaie
\end{itemize}
\end{frame}

\subsection{Seconde modélisation}

\begin{frame}[fragile] \frametitle{Seconde modélisation : objectifs et principes}
\begin{itemize}
\item Objectifs
	\begin{itemize}
	\item Comparer les méthodes de \textit{PoW} et \textit{PoS}
	\item Etudier la vitesse de remplissage de la \textit{blockchain} et la rapidité de diffusion au sein du réseau d'utilisateurs en fonction des paramètres 
		\begin{itemize}
		\item nb d'utilisateurs, nb de tours de boucle, difficulté
		\end{itemize}
	\end{itemize}
\item Principes
	\begin{itemize}
	\item $n$ utilisateurs regroupés dans une liste
	\item Chaque tour, pour chaque utilisateur : 
		\begin{itemize}
		\item PoW ou PoS 
		\item Comparaison de la \textit{blockchain} avec 10 utilisateurs choisis au hasard
		\end{itemize}
	\end{itemize}
\end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame} \frametitle{Conclusion}
\begin{itemize}
\item Premiers résultats
	\begin{itemize}
		\item Diffusion plus importante si la difficulté est plus élevée
	\item Dans notre modèle, si la puissance d'un seul utilisateur est plus élevée que celle des autres, sa \textit{blockchain} a tendance à prédominer
		\end{itemize}
	\item Suites du travail 
		\begin{itemize}
		\item Comparer la \textit{Proof of Work} et la \textit{Proof of Stake}
		\item Quels sont les paramètres de PoS les plus économes tout en conservant la sécurité de la PoW
		\end{itemize}
\end{itemize} 
\end{frame}
