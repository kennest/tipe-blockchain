\begin{frame} \frametitle{Fonctionnement général}
\begin{itemize}
\item Fonctionnement multi-agents, en effectuant de multiples itérations sur le réseau
\item Au départ, un agent a une information
\item Itération :
	\begin{itemize}
	\item Parcours des agents un à un
	\item Si \emph{normal} : passe son information à tous les voisins qui ne la possèdent pas
	\item Si \emph{attaquant} : envoie à tous ses voisins qui ne possèdent pas encore l'information une information fausse
	\end{itemize}
\end{itemize}
\end{frame}

\begin{frame} \frametitle{Les types de réseaux - En étoile}
\begin{itemize}
\item Étoile : Un noeud est connecté avec tous les autres. C'est l'architecture d'un système client-serveur.
\item Ce type de réseau est assez facile à étudier.
\end{itemize}

\begin{frame} \frametitle{Les types de réseaux - Homogène}
\begin{itemize}
\item \emph{Homogène} : les degrés sortants de chaque noeud sont égaux
\item nous n'étudierons que des réseaux connexes (strictement dans le cas orienté)

\begin{frame} \frametitle{Les types de réseaux - Reseau invariant d'échelle}
% Images !
\begin{itemize}
\item \emph{Scale-free}, ou invariant d'échelle : Le nombre d'arêtes par noeud suit une loi de puissance : c'est-à-dire que la probabilité qu'un noeud ait $k$ voisins est $k^{-\gamma}$ Le réseau Internet est de ce type.
\item Nous utiliserons l'agorithme de Barabási-Albert, pour lequel $\gamma = 3$.
\end{itemize}

\subsection{Sur un réseau homogène}

\begin{frame} \frametitle{Variation du nombre de noeuds}

\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-10-3-4.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-1.png} 

\begin{itemize}
\item Nombre de réponses fausses selon le nombre d'attaquants
\item Croissant, mais la dérivée seconde est négative : plus il y a d'attaquants, et moins l'action d'en ajouter un nouveau est significative
\end{itemize}
\end{frame}



\begin{frame} \frametitle{Variation du nombre d'arêtes}
\includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-3-1.png} \includegraphics[scale=0.3]{../../../resultats/atkaleat/atkaleat-50-20-1.png} 

\begin{itemize}
	\item Variation du nombre de tunnels par agent
	\item Courbe se divise en deux parties, la seconde affines
	\item A l'arrivée sur la dernière partie, les seuls agents ayant des informations vrais sont voisins de l'émetteur
\end{itemize}
\end{frame}

\section{Conclusion}

\begin{frame} \frametitle{Conclusion}
\begin{itemize}
	\item Résultats
	\begin{itemize}
		\item Courbes non linéaires, croissantes
		\item Pour les graphes homogènes, division en deux parties affines
		\item Réseaux invariants d'échelle : présence de paliers
	\end{itemize}
	\item En pratique
	\begin{itemize}
		\item Le réseau en étoile est peu vulnérable, si le noeud central est non-attaquant
		\item Réseau homogène difficile à mettre en place
		\item Dans le réseau scale-free, on a supposé que les noeuds ayant une forte connectivité ont la même probabilité d'être attaquants que les autres. Or en pratique, ce sont souvent des noeuds plus sécurisés.
	\end{itemize}
\end{itemize} 
\end{frame}
